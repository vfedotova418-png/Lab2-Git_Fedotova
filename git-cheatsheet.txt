git version - отображает текущую версию git
git config --global user.name "ваш_ник" - устанавливает имя пользователя, применяемое ко всем git-репозиториям на компьютере
git config --global user.email "ваша_почта" - устанавливает почту пользователя, применяемое ко всем git-репозиториям на компьютере
cd [название папки] - переход в указанную папку
mkdir [название папки] - создаёт пустую папку
pwd - отображает путь к текущей папке
git init -b main - создаёт git-репозиторий и указывает имя основной ветки
git remote add origin ссылка_на_репозиторий - подключает удалённый репозиторий к локальному
git remote -v - отображает адреса удалённого репозитория
dotnet new console -o Program --use-program-main - создаёт новый шаблон консольного приложения, указывает имя папки где он будет создан, и создаёт классическую точку входа Main
git status - отображает состояние репозитория
git add . - добавляет все файлы в индекс
git commit -m "сообщение коммита" - создаёт коммит с написаным сообщением
git log - отображает историю коммитов в текущей ветке
git push - отправляет все коммиты в удалённый репозиторий
git log --oneline - отображает историю коммитов в компактном формате
git show ваш_хеш - отображает детальную информацию о конкретном коммите
git checkout -b [название ветки], git switch -c [название ветки] - создание новой ветки и переключение на неё
git checkout [название ветки], git switch [название ветки], git branch [название ветки] - создаёт ветку не переходя в неё
git branch - отображает все ветки
git switch - - переход на предыдущую ветку
git branch -d [название ветки] - удаляет ветку
git pull - обновление ветки локального репозитория
git fetch --prune - отображает актуальное состояние сервера
